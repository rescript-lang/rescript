%graphql(`
  query Site {
    site {
      siteMetadata {
        title
        description
        siteUrl
      }
    }
  }
`{taggedTemplate: false})

module Form = %form(type input = {
  name: string,
  email: string,
  message: string,
  formName: string
}type output = inputlet validators = {name: {strategy: OnFirstBlur, validate: (({name: name, _}) => switch name {
  | "" => Error("Name is required.")
  | name => Ok(name)
})}, email: {strategy: OnFirstBlur, validate: (({email: email, _}) => switch email {
  | "" => Error("Email is required.")
  | email => Ok(email)
})}, message: {strategy: OnFirstBlur, validate: (({message: message, _}) => switch message {
  | "" => Error("Message is required.")
  | message => Ok(message)
})}, formName: {strategy: OnSubmit, validate: (({formName: formName, _}) => Ok(formName))}})
