// empty dict
let x = Primitive_dict.make([])

// empty dict with inside comment
let x = Primitive_dict.make([])

// inside comment
// one value
let x = Primitive_dict.make([("foo", "bar")])

// two values
let x = Primitive_dict.make([("foo", "bar"), ("bar", "baz")])

let baz = "foo"

let x = Primitive_dict.make([("foo", "bar"), ("bar", "baz"), ("baz", baz)])

// multiline
let x = Primitive_dict.make([("foo", "bar"), ("bar", "baz"), ("baz", baz)])

// comments
let x /* foo */  = Primitive_dict.make([("foo", "bar")])

let x /* foo */  = Primitive_dict.make([("foo", "bar")])

let x /* foo */  = Primitive_dict.make([("foo", "bar")])

let x = Primitive_dict.make([
// foo
("foo", "bar"), 
// bar
("bar", "baz"), 
// baz
("baz", baz)])

let x = Primitive_dict.make([("foo" // foo
, "bar"), ("bar" // bar
, "baz"), ("baz" // baz
, baz)])

let x = Primitive_dict.make([("foo" /* foo */ , "bar"), ("bar" /* bar */ , "baz"), ("baz" /* bar */ , baz)])

let x = Primitive_dict.make([
/* foo */
("foo", "bar"), 
/* bar */
("bar", "baz"), 
/* bar */
("baz", baz)])

let x = Primitive_dict.make([("foo" /* foo */ , "bar"), ("bar" /* bar */ , "baz"), ("baz" /* bar */ , baz)])
