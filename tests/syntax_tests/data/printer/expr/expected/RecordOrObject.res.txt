type rx = {
  x: int
}

type ry = {
  y: string
}

type rxi = {
  ...: rx,
  i: int
}

type rxy // this is a record
 = {
  ...: rx,
  ...: ry
}

let vxy: rxy = {x: 10, y: "abc"}

let xxi: rxi = {x: 10}

type ox = {"x": int}

type oy = {"y": int}

type oxz = {...ox, "z": int}

 = {
  ...: ox,
  ...: oy
